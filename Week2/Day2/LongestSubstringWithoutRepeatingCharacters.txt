class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        unordered_map<char,int> freq;
        int maxl=0,left=0;
        for(int right=0;right<s.length();right++){
            char ch=s[right];
            freq[ch]+=1; //freq[ch]++;
            while(freq[ch]>1){
                char leftch=s[left];
                freq[leftch]-=1;
                left++;
            }
            maxl=max(maxl,right-left+1);
        }
        return maxl;
    }
};
