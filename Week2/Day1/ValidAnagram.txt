class Solution {
public:
    bool isAnagram(string s, string t) {
        if(s.size()!=t.size()) return false;
        unordered_map<char,int> count;
        for(char c:s){
            count[c]+=1;
        }
        for(char c:t){
            if(count[c]==0||count.find(c)==count.end()) return false;
            count[c]-=1;
        }
        return true;
    }
};
