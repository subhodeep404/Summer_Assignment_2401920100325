class Solution {
public:
    vector<int> twoSum(vector<int>& ns, int target) {
        unordered_map<int,int> map1;
        for(int i=0;i<nums.size();i++){
            int n=nums[i];
            int rem=target-n;
            if(map1.find(rem)!=map1.end()){
                return {map1[rem],i};
            }
            map1[n]=i;
        }
        return{-1,-1};
    }
};
